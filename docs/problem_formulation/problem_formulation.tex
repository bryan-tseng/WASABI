\subsection*{There are a few ways we can formulate the question}
\begin{enumerate}
    \item Similar to Draelos 2020 first identify the binary connections using group testing, then use the identified connection to fit a weighted model.
    \item Directly use a linear-Poisson model to identify weights using group testing. (Works on the assumption that connections are linear)
    \item If not linear, use NN or some other nonlinear approaches.
\end{enumerate}
We will focus on (2) first, with fixed continuous stimulation parameters (e.g., current, frequency, pulse width).
\subsection*{Poisson MLE for rate estimation}
\[
\begin{aligned}
\text{Let } & x_i \in \{0,1\} 
&& \text{be the selection of stimulation channels } i,
\quad i = 1,\ldots,N \\
\text{Let } & r \in \mathbb{R}^M
&& \text{be the recorded response from } M \text{ recording channels}
\end{aligned}
\]
We assume a linear-Poisson model where the response is generated as:
\[
\begin{aligned}
r_j &\sim \text{Poisson}(\lambda_j), \text{where } \lambda_j 
= \sum_{k=1}^K D_{jk}\, z_k + b_j^{prestim} \\
\lambda_{j}  &:\ \text{firing rate of channel j} \\
D_{jk} &:\ \text{weight from latent variable } k \text{ to recording channel } j \\
z_{k}  &:\ \text{latent variable k (Assumes rate to stim-channel relationship is low-rank)} \\
K      &:\ \text{total number of latent variables} \\
b_j^{prestim}    &:\ \text{baseline firing rate of recording channel } j \text{ identified from the prestim period}
\end{aligned}
\]

We further define the matrix $W = [w_{ki}]$ which identifies the mapping from the stimulation pattern $x_{i}$ to latent variable $z_{k}$
\[
\begin{aligned}
z_k =  \sum_{i=1}^NW_{ki}\, x_i\\
\end{aligned}
\]

Given a set of stimulation patterns and recorded responses $\{(x_t, r_t)\}_{t=1}^T$, where $T$ is the number of trials. The Poisson likelihood is
\begin{equation}
    \mathcal{L}(r; \lambda) = \prod_{t=1}^T \prod_{j=1}^M \frac{e^{-\lambda_{j, t}} (\lambda_{j, t})^{r_{j, t}}}{r_{j, t}!}
\end{equation}

{\centering 
where $\lambda_j = \sum_{i=1}^K D_{ji} W_{ki}x_i + b_j^{prestim}$\par}
This gives the log-likelihood:
\begin{equation}
    \log \mathcal{L}(r; \lambda) = \sum_{t=1}^T \sum_{j=1}^M \left( -\lambda_j + r_j \log \lambda_j - \log r_j! \right)
\end{equation}
Constraints:
\begin{itemize}
    \item $\lambda_{j} \geq 0$ (non-negative firing rate) or just use softplus link function, which cancels out with log?
    \item $b_{j} \geq 0$ (non-negative baseline firing rate)
\end{itemize}

\subsection*{Minimizing the posterior covariance of W (Gaussian)}
For adaptive selection of stimulation patterns, at each trial $t$, we want to select $\textbf{x}_{t}$ that maximizes the expected information gain about \textbf{W}. Equivalently, this would be selecting $\textbf{x}_{t}$ that minimizes the determinant of the covariance of \textbf{W} at time t. We separate out the Poisson MLE rate estimation and identification of the optimal $\textbf{W}^*$ in our formulation. This formulation could be good for control-theoretic implementations later too.

\begin{equation}
    \lambda_t =\textbf{D}\textbf{W}\textbf{x}_t+\textbf{b}^{prestim}_{t}
\end{equation}

Here, we are trying to find the optimal $\textbf{W}^*$, where $\lambda_t$, $\textbf{x}_t$, and $\textbf{b}^{prestim}_{t}$ are known.
\[
\begin{aligned}
\text{Let } & \Sigma_{t-1}
&& \text{be the covariance matrix of \textbf{W} before trial $t-1$}\\
\text{Let } & \beta
&& \text{be how we scale our data relative to our prior (could be $1/n$)}
\end{aligned}
\]

In terms of precision, we can formulate the Gaussian case as:
\begin{equation}
    \Sigma_{t}^{-1} = \Sigma_{t-1}^{-1} + \beta \; \mathbb{E}[\lambda_t\mid \textbf{x}_{t}] \mathbb{E}[\lambda_t\mid \textbf{x}_{t}]^T
\end{equation}

Take the determinant
\[
    \text{det}\;\Sigma_{t}^{-1} = \text{det}(\Sigma_{t-1}^{-1} + \beta \; \mathbb{E}[\lambda_t\mid \textbf{x}_{t}] \mathbb{E}[\lambda_t\mid \textbf{x}_{t}]^T)
\]

using the matrix inversion lemma:
\[
    \text{det}(\textbf{A} + \textbf{u}\textbf{v}^T) = (1 + \textbf{v}^T\textbf{A}^{-1}\textbf{u})\text{det}(\textbf{A})
\]
\[
    \text{det}(\Sigma_{t-1}^{-1} + \beta \; \mathbb{E}[\lambda_t\mid \textbf{x}_{t}] \mathbb{E}[\lambda_t\mid \textbf{x}_{t}]^T) =  (1+ \beta \; \mathbb{E}[\lambda_t\mid \textbf{x}_{t}]^T \Sigma_{t-1} \mathbb{E}[\lambda_t\mid \textbf{x}_{t}])\det(\Sigma_{t-1}^{-1})
\]

Since we minimize over $x_{t}$, the objective function is therefore:
\[
\begin{aligned}
    \arg\max_{x}\; \det\big(\Sigma_{t}^{-1}\big)
    &= \arg\max_{x}\;  \big(1+ \beta \,\mathbb{E}[\lambda_t\mid \textbf{x}_{t}]^{\top}\Sigma_{t-1}\mathbb{E}[\lambda_t\mid \textbf{x}_{t}]\big)\,\det(\Sigma_{t-1}^{-1})\\
    &= \arg\max_{x}\; \beta \,\mathbb{E}[\lambda_t\mid \textbf{x}_{t}]^{\top}\Sigma_{t-1}\mathbb{E}[\lambda_t\mid \textbf{x}_{t}]
\end{aligned}
\]

We can further expand out:
\[
\begin{aligned}
    \mathbb{E}[\lambda_t\mid \textbf{x}_{t}]=\hat{\textbf{D}}_{t-1}\hat{\textbf{W}}_{t-1}\textbf{x}_{t}+\textbf{b}_{t}^{prestim}
\end{aligned}
\]
where $\hat{\textbf{D}}_{t-1}$ and $\hat{\textbf{W}}_{t-1}$ are estimated from timepoints $i = 1,\ldots, t-1$
and $\textbf{b}_{t}^{prestim}$ is baseline firing rate identified from the prestim period.

\section{Random Sampling Stopgap}

\begin{itemize}
    \item Total configurations: $N$
    \item Take $N$ samples: stimulate groups of 2 sites (one row). 
    \item For 128 sites, this is 64 rows. 
    \item Each stim randomly sample 20\%: 12 rows (can revisit this number)
    \item Collect all the data and responses and see if we can predict in a leave-one-out approach (use all but one response). 
    \item Benefit of random sampling is that we can collect now before Nike's implant comes out.
\end{itemize}
