\subsection*{There are a few ways we can formulate the question}
\begin{enumerate}
    \item Similar to Draelos 2020 first identify the binary connections using group testing, then use the identified connection to fit a weighted model.
    \item Directly use a linear-Poisson model to identify weights using group testing. (Works on the assumption that connections are linear)
    \item If not linear, use NN or some other nonlinear approaches.
\end{enumerate}
We will focus on (2) first, with fixed continuous stimulation parameters (e.g., current, frequency, pulse width).

\subsection*{Simplified formulation}
\begin{align*}
\mathbf{r} &\sim \mathrm{Poisson}(\boldsymbol{\lambda}) \\
\boldsymbol{\lambda}^{\mathrm{stim}} &= \mathbf{Cs} + \boldsymbol{\lambda}^{\mathrm{control}} \\
\mathbf{C} &= \mathbf{DW}_{\text{low-rank}}
\end{align*}
\begin{align*}
\mathbf{r} &:\ \text{Recorded responses (spikes)}\\
\boldsymbol{\lambda}  &:\ \text{Firing rates} \\
\mathbf{C} &:\ \text{Mapping from stimulation channels to firing rates} \\
\mathbf{s} &:\ \text{Selection of stimulation channels}\\
\mathbf{D} &:\ \text{Mapping from latent variables to recorded responses} \\
\mathbf{W}_{\text{low-rank}} &:\ \text{Low rank mapping from stimulation channels to latent variables}
\end{align*}

\begin{itemize}
\item $\mathbf{W}$ is unknown.
\item At time t-1, we want to find $\mathbf{s}_t$ that maximizes the information about $\mathbf{W}$, given $\mathbf{r}_{1:t-1}$ and $\mathbf{s}_{1:t-1}$
\end{itemize}

\subsection*{Ridge regression for learning W at time t}
If we assume we know $D_t$ at time $t$, we estimate $\mathbf{W}_t$ using a ridge least-squares objective:
\begin{align*}
\hat{\mathbf{W}}_t &= \arg\min_{\mathbf{W}}
||\mathbf{Z-WS}||_F^2 + \beta||\mathbf{W}||_F^2
\end{align*}
where $\beta$ is the weight of the ridge prior. We can expanding the Frobenius norm using trace:
\begin{align*}
\mathcal{L}(\hat{\mathbf{W}}_t) &= tr[(\mathbf{Z-WS})(\mathbf{Z-WS})^{\top}]+\beta \,tr(\mathbf{WW}^{\top})\\
\mathcal{L}(\hat{\mathbf{W}}_t)&=tr(\mathbf{ZZ}^{\top})
- 2\,tr(\mathbf{ZS^{\top}W^{\top}}) + tr(\mathbf{WSS^{\top}W^{\top}})
+ \beta \,tr(\mathbf{WW}^{\top})
\end{align*}
Keep terms relative to $\mathbf{W}_t$
\begin{align*}
\mathcal{L}(\hat{\mathbf{W}}_t)= -2\,\mathrm{tr}(\mathbf{ZS}^{\top}\mathbf{W}^{\top})
+ \mathrm{tr}(\mathbf{WSS^{\top}W^{\top}})
+ \beta \,\mathrm{tr}(\mathbf{WW}^{\top})
\end{align*}
Taking derivative and set to zero:

\begin{align*}
\frac{\partial \mathcal{L}}{\partial \mathbf{W}} &= -2\,\mathbf{ZS}^{\top}
+ 2\,\mathbf{WSS}^{\top}
+ 2\beta\,\mathbf{W}\\
0 &= -2\,\mathbf{ZS}^{\top} + 2\,\mathbf{WSS}^{\top}+ 2\beta\,\mathbf{W}\\
\mathbf{WSS}^{\top} &+ \beta \mathbf{W} = \mathbf{ZS}^{\top}\\
\mathbf{W}(\mathbf{SS}^{\top} &+ \beta \mathbf{I}) = \mathbf{ZS}^{\top}
\end{align*}
\begin{align*}
\boxed{\mathbf{W}=\mathbf{ZS}^{\top}(\mathbf{SS}^{\top}+\beta \mathbf{I})^{-1}}
\end{align*}

\subsection*{Generative model for perturbation}
\begin{align*}
(\boldsymbol{\lambda}_{\text{stim}}-\boldsymbol{\lambda}_{\text{control}})=\mathbf{D}(\mathbf{WS}+\boldsymbol{\xi})+\boldsymbol{\eta}
\end{align*}
Note:
\begin{itemize}
\item Current loss is written for estimating $\mathbf{W}_t$
\item $\mathbf{D}$ - learned from control? or learn jointly with trials t (Alternateing or SVD)?
\item $\boldsymbol{\xi}_t$ is perturbation noise (e.g.\ mismatch between intended and actual stim-response effects) - hyperparameter, match data?
\item $\boldsymbol{\eta}$: observation noise - hyperparameter, match data?
\item Columns of $\mathbf{W}$ have unit variance
\item Ridge prior should scale as $\beta \propto \frac{1}{n}$ so in low-trial regime solution is not degenerate and high trial regime data info is respected.
\end{itemize}

\subsection*{Dynamics model}

Instead of assuming an instantaneous mapping $\mathbf{W}$, we model stimulation as a perturbation to a latent dynamical system through the mapping $\mathbf{B}$. We can also just combine rate estimation with the observation equation (also dimensionality reduction)
\paragraph{State equation}
\begin{align*}
    \mathbf{x}_0 &\sim \mathcal{N}(\boldsymbol{\mu}_0, \boldsymbol{\Sigma}_0) \\
    \mathbf{x}_t &= \mathbf{A}\mathbf{x}_{t-1} + \mathbf{B}\mathbf{s}_t + \boldsymbol{\epsilon}_t, \quad \boldsymbol{\epsilon}_t \sim \mathcal{N}(\mathbf{0}, \mathbf{Q}) \\
    \mathbf{y}_t &\sim \text{Poisson}\!\left(\text{softplus}\!\left(\mathbf{C}\mathbf{x}_t + \mathbf{d} + \mathbf{D}\mathbf{s}_t\right)\right)
\end{align*}
where:
\begin{itemize}
    \item $\mathbf{A}\in\mathbb{R}^{K\times K}$ Dynamics operator
    \item $\mathbf{B}\in\mathbb{R}^{K\times N}$: Stimulus-to-latent mapping
    \item $\mathbf{C}\in\mathbb{R}^{N\times K}$: Observation (loading) matrix
    \item $\mathbf{D}\in\mathbb{R}^{N\times N}$: Direct ch-to-ch effect that is a diagonal matrix (TODO Might need a kernel for delayed effects)
    \item $\mathbf{d}\in\mathbb{R}^{N}$: per-channel mean log firing rate
    \item $\mathbf{Q}\in\mathbb{R}^{K\times K}$: latent process noise covariance
    \item $\mathbf{s}_t\in\mathbb{R}^{N}$: stimulus input at time $t$ (Control trials just have $\vec{\mathbf{0}}$)
    \item $\mathbf{y}_t\in\mathbb{R}^{N}$: binned spike counts
    \item $K \ll N$ (low-rank assumption)
\end{itemize}

\subsection*{Fitting using EKF-style integral and SGD (Ignore Ds for now)}
The marginal likelihood is:
\begin{align*}
    \log p(\mathbf{y}_{1:t};\boldsymbol{\theta})
        &= \log \int p(\mathbf{y}_{1:t} \mid \mathbf{x}_{1:t};\boldsymbol{\theta})\,
                       p(\mathbf{x}_{1:t};\boldsymbol{\theta})\, d\mathbf{x}_{1:t}\\
\mathbf{y}_t &\sim \text{Poisson}\!\left(\text{softplus}\!\left(\mathbf{C}\mathbf{x}_t + \mathbf{d}\right)\right)\\
\boldsymbol{\theta} &= \{\mathbf{A, B, C, d, \mu_0, \Sigma_0}\}
\end{align*}

Because the Poisson emission $p(\mathbf{y}_{1:T} \mid \mathbf{x}_{1:T};\boldsymbol{\theta})$
is non-Gaussian, the integral is intractable. Need to approximate it by linearizing
the emission around $\hat{\mathbf{x}}_t$, the current filtered mean (latent states), yielding a Gaussian approximation to the predictive distribution
\begin{align*}
p(\mathbf{y}_t \mid \mathbf{y}_{1:t-1};\boldsymbol{\theta})
\approx \mathcal{N}(\boldsymbol{\lambda}_{t|t-1},\, \mathbf{S}_t)
\end{align*}

where
\begin{align*}
\boldsymbol{\lambda}_{t|t-1} = \text{softplus}\!\left(\mathbf{C}\hat{\mathbf{x}}_{t|t-1} + \mathbf{d}\right)
\end{align*}

and $\mathbf{S}_t$ is the estimated covariance.

This gives the approximate marginal likelihood
\begin{align*}
    \mathcal{L}(\boldsymbol{\theta})
        \approx \sum_{t=1}^{T} \log \mathcal{N}(\mathbf{y}_t;\,
                \boldsymbol{\lambda}_{t|t-1},\, \mathbf{S}_t)
\end{align*}

which we can maximize directly via SGD.

\paragraph{Emission Jacobian}
For $\boldsymbol{\lambda}_t = \text{softplus}(\mathbf{C}\hat{\mathbf{x}}_t + \mathbf{d})$, we have the Jacobian (log-rate with respect to the latent state) is:
\begin{align*}
    \mathbf{H}_t &= \frac{\partial \boldsymbol{\lambda}_t}{\partial \mathbf{x}_t}\Bigg|_{\hat{\mathbf{x}}_t}
                 = \text{diag}(\sigma(\mathbf{C}\hat{\mathbf{x}}_t + \mathbf{d}))\,\mathbf{C}
                 \in \mathbb{R}^{N \times K}
\end{align*}
where $\sigma(\cdot)$ is the sigmoid function (derivative of softplus). We calculate the Emission Jacobian $\mathbf{H}_t$ to linearize the
non-Gaussian Poisson emission and plug into the Kalman equations. This propagates the Gaussian estimations of latent states $\mathbf{x}_t$ one time step forward each.
\clearpage
\paragraph{EKF forward pass prediction update}
\begin{align*}
    \hat{\mathbf{x}}_{t|t-1} &= \mathbf{A}\hat{\mathbf{x}}_{t-1|t-1} + \mathbf{B}\mathbf{s}_t \\
    \mathbf{P}_{t|t-1}       &= \mathbf{A}\mathbf{P}_{t-1|t-1}\mathbf{A}^\top + \mathbf{Q} \\
    \mathbf{S}_t             &= \mathbf{H}_t \mathbf{P}_{t|t-1} \mathbf{H}_t^\top
                                + \text{diag}(\boldsymbol{\lambda}_{t|t-1})^{-1} \\
    \mathbf{K}_t             &= \mathbf{P}_{t|t-1}\mathbf{H}_t^\top \mathbf{S}_t^{-1} \\
    \hat{\mathbf{x}}_{t|t}   &= \hat{\mathbf{x}}_{t|t-1}
                                + \mathbf{K}_t(\mathbf{y}_t - \boldsymbol{\lambda}_{t|t-1}) \\
    \mathbf{P}_{t|t}         &= (\mathbf{I} - \mathbf{K}_t \mathbf{H}_t)\mathbf{P}_{t|t-1}
\end{align*}
where
\begin{itemize}
    \item $\hat{\mathbf{x}}_{t|t-1} \in \mathbb{R}^{K}$: predicted latent state mean
    \item $\hat{\mathbf{x}}_{t|t} \in \mathbb{R}^{K}$: filtered latent state mean
    \item $\mathbf{P}_{t|t-1} \in \mathbb{R}^{K \times K}$: predicted latent state covariance
    \item $\mathbf{P}_{t|t} \in \mathbb{R}^{K \times K}$: filtered latent state covariance
    \item $\mathbf{S}_t \in \mathbb{R}^{N \times N}$: innovation covariance
    \item $\mathbf{K}_t \in \mathbb{R}^{K \times N}$: Kalman gain
    \item $(\mathbf{y}_t - \boldsymbol{\lambda}_{t|t-1})$: innovation (residual between observed spikes and predicted rate)
    \item $\mathbf{Q} \in \mathbb{R}^{K \times K}$: latent process noise covariance
    \item $\text{diag}(\boldsymbol{\lambda}_{t|t-1})^{-1}$: Poisson observation noise (variance equals mean)
\end{itemize}

\paragraph{EKF marginal log-likelihood}
With this we have the defined objective with the approximated $\boldsymbol{\lambda}_{t|t-1}, \mathbf{S}_t$
\begin{align*}
    \mathcal{L}(\boldsymbol{\theta})
        &\approx \sum_{t=1}^{T} \log \mathcal{N}(\mathbf{y}_t;\,
                 \boldsymbol{\lambda}_{t|t-1},\, \mathbf{S}_t)
\end{align*}
For missing data, we zero out the corresponding terms in the likelihood.

\paragraph{SGD parameter update}
Parameters are updated by differentiating through the EKF pass:
\begin{align*}
    \boldsymbol{\theta}^{(\tau+1)}
        &= \boldsymbol{\theta}^{(\tau)}
           - \eta_\tau \,\nabla_{\boldsymbol{\theta}}\,
             \bigl[-\mathcal{L}(\boldsymbol{\theta}^{(\tau)})\bigr]
\end{align*}
where $\eta_\tau$ is the effective learning rate defined by the Adam optimizer. More specifically:
\begin{align*}
    \mathbf{m}_{\tau} &= \beta_1 \mathbf{m}_{\tau-1}
                         + (1-\beta_1)\,\nabla_{\boldsymbol{\theta}}
                         \bigl[-\mathcal{L}(\boldsymbol{\theta}^{(\tau)})\bigr]
                         \tag{momentum} \\
    \mathbf{v}_{\tau} &= \beta_2 \mathbf{v}_{\tau-1}
                         + (1-\beta_2)\,\nabla_{\boldsymbol{\theta}}
                         \bigl[-\mathcal{L}(\boldsymbol{\theta}^{(\tau)})\bigr]^2
                         \tag{RMSprop} \\
    \hat{\mathbf{m}}_{\tau} &= \frac{\mathbf{m}_{\tau}}{1-\beta_1^\tau}, \qquad
    \hat{\mathbf{v}}_{\tau}  = \frac{\mathbf{v}_{\tau}}{1-\beta_2^\tau}
                         \tag{bias correction} \\
    \boldsymbol{\theta}^{(\tau+1)}
                      &= \boldsymbol{\theta}^{(\tau)}
                         - \frac{\alpha}{\sqrt{\hat{\mathbf{v}}_{\tau}}+\epsilon}\,
                         \hat{\mathbf{m}}_{\tau}
                         \tag{parameter update}
\end{align*}
where $\mathbf{m}_\tau$ and $\mathbf{v}_\tau$ are the first and second moment
estimates, $\beta_1 = 0.9$, $\beta_2 = 0.999$, $\epsilon = 10^{-8}$ are the
default hyperparameters, and $\alpha$ is the base learning rate.

and gradients are computed via automatic differentiation in JAX.
The full parameter set is:
\begin{align*}
    \boldsymbol{\theta} &= \{\mathbf{A},\,\mathbf{B},\,\mathbf{C},\,
                             \mathbf{d},\,\mathbf{Q},\,\boldsymbol{\mu}_0,\,\boldsymbol{\Sigma}_0\}
\end{align*}

\subsection*{Minimizing the posterior covariance of W (Gaussian)}
For adaptive selection of stimulation patterns, at each trial $t$, we want to select $\mathbf{s}_{t}$ that maximizes the expected information gain about $\mathbf{W}$. Equivalently, this would be selecting $\mathbf{s}_{t}$ that minimizes the determinant of the covariance of $\mathbf{W}$ at time t. We separate out the Poisson MLE rate estimation and identification of the optimal $\mathbf{W}^*$ in our formulation. This formulation could be good for control-theoretic implementations later too.

\begin{equation}
    \boldsymbol{\lambda}_t =\mathbf{DWs}_t+\boldsymbol{\lambda}^{control}_{t}
\end{equation}

Here, we are trying to find the optimal $\mathbf{W}^*$, where $\boldsymbol{\lambda}$, $\mathbf{s}_t$, and $\boldsymbol{\lambda}^{control}_{t}$ are known.
\begin{align*}
    \text{Let } & \Sigma_{t-1}
    && \text{be the covariance matrix of $\mathbf{W}$ before trial $t-1$}\\
    \text{Let } & \beta
    && \text{be how we scale our data relative to our prior (could be $1/n$)}
\end{align*}

In terms of precision, we can formulate the Gaussian case as:
\begin{equation}
    \Sigma_{t}^{-1} = \Sigma_{t-1}^{-1} + \beta \; \mathbb{E}[\boldsymbol{\lambda} _t\mid \mathbf{s}_{t}] \mathbb{E}[\boldsymbol{\lambda} _t\mid \mathbf{s}_{t}]^T
\end{equation}

Take the determinant
\[
    \text{det}\;\Sigma_{t}^{-1} = \text{det}(\Sigma_{t-1}^{-1} + \beta \; \mathbb{E}[\boldsymbol{\lambda} _t\mid \mathbf{s}_{t}] \mathbb{E}[\boldsymbol{\lambda} _t\mid \mathbf{s}_{t}]^T)
\]

using the matrix inversion lemma:
\[
    \text{det}(\mathbf{A} + \mathbf{uv}^T) = (1 + \mathbf{v}^T\mathbf{A}^{-1}\mathbf{u})\text{det}(\mathbf{A})
\]
\[
    \text{det}(\Sigma_{t-1}^{-1} + \beta \; \mathbb{E}[\boldsymbol{\lambda} _t\mid \mathbf{s}_{t}] \mathbb{E}[\boldsymbol{\lambda} _t\mid \mathbf{s}_{t}]^T) =  (1+ \beta \; \mathbb{E}[\boldsymbol{\lambda} _t\mid \mathbf{s}_{t}]^T \Sigma_{t-1} \mathbb{E}[\boldsymbol{\lambda} _t\mid \mathbf{s}_{t}])\det(\Sigma_{t-1}^{-1})
\]

Since we minimize over $\mathbf{s}_{t}$, the objective function is therefore:
\begin{align*}
    \arg\max_{\mathbf{s}}\; \det\big(\Sigma_{t}^{-1}\big)
    &= \arg\max_{\mathbf{s}}\;  \big(1+ \beta \,\mathbb{E}[\boldsymbol{\lambda} _t\mid \mathbf{s}_{t}]^{\top}\Sigma_{t-1}\mathbb{E}[\boldsymbol{\lambda} _t\mid \mathbf{s}_{t}]\big)\,\det(\Sigma_{t-1}^{-1})\\
    &= \arg\max_{\mathbf{s}}\; \beta \,\mathbb{E}[\boldsymbol{\lambda} _t\mid \mathbf{s}_{t}]^{\top}\Sigma_{t-1}\mathbb{E}[\boldsymbol{\lambda} _t\mid \mathbf{s}_{t}]
\end{align*}

We can further expand out:
\begin{align*}
    \mathbb{E}[\boldsymbol{\lambda} _t\mid \mathbf{s}_{t}]=\hat{\mathbf{D}}_{t-1}\hat{\mathbf{W}}_{t-1}\mathbf{s}_{t}+\boldsymbol{\lambda}_{t}^{control}
\end{align*}
where $\hat{\mathbf{D}}_{t-1}$ and $\hat{\mathbf{W}}_{t-1}$ are estimated from timepoints $i = 1,\ldots, t-1$
and $\boldsymbol{\lambda}_{t}^{control}$ is firing rate identified from control reaches.

\subsection*{Poisson MLE for rate estimation}
\begin{align*}
\text{Let } & s_i \in \{0,1\} 
&& \text{be the selection of stimulation channels } i,
\quad i = 1,\ldots,N \\
\text{Let } & r \in \mathbb{R}^M
&& \text{be the recorded response from } M \text{ recording channels}
\end{align*}

We assume a linear-Poisson model where the response is generated as:
\begin{align*}
r_j &\sim \text{Poisson}(\lambda_j), \text{where } \lambda_j 
= \sum_{k=1}^K D_{jk}\, z_k + \lambda_j^{control} \\
\lambda_{j}  &:\ \text{firing rate of channel j} \\
D_{jk} &:\ \text{weight from latent variable } k \text{ to recording channel } j \\
z_{k}  &:\ \text{latent variable k (Assumes rate to stim-channel relationship is low-rank)} \\
K      &:\ \text{total number of latent variables} \\
\lambda_j^{control}    &:\ \text{firing rate of recording channel } j \text{ identified from control reaches}
\end{align*}

We further define the matrix $W = [w_{ki}]$ which identifies the mapping from the stimulation pattern $s_{i}$ to latent variable $z_{k}$

\begin{align*}
z_k =  \sum_{i=1}^N W_{ki}\, s_i\\
\end{align*}

Given a set of stimulation patterns and recorded responses $\{(\mathbf{s}_t, \mathbf{r}_t)\}_{t=1}^T$, where $T$ is the number of trials. The Poisson likelihood is
\begin{equation}
    \mathcal{L}(r; \lambda) = \prod_{t=1}^T \prod_{j=1}^M \frac{e^{-\lambda_{j, t}} (\lambda_{j, t})^{r_{j, t}}}{r_{j, t}!}
\end{equation}

{\centering 
where $\lambda_{j,t} = \sum_{k=1}^K \sum_{i=1}^N D_{jk}\, W_{ki}\, s_{i,t} + \lambda_{j,t}^{control}$\par}
This gives the log-likelihood:
\begin{equation}
    \log \mathcal{L}(r; \lambda) = \sum_{t=1}^T \sum_{j=1}^M ( -\lambda_{j,t} + r_{j,t} \log \lambda_{j,t} - \log (r_{j,t}!))
\end{equation}
Constraints:
\begin{itemize}
    \item $\lambda_{j} \geq 0$ (non-negative firing rate) or just use softplus link function, which cancels out with log?
    \item $\lambda_{j} \geq 0$ (non-negative control firing rate)
\end{itemize}
